    std::vector merge_sort (int n, std::vector array){
        int size = n;
    
        if (size <= 1){
            return array;
        }
    
        int mid = size/2;
        std::vector left_array(array.begin(), array.begin() + mid);
        std::vector right_array(array.begin() + mid, array.end());
    
        return merge(merge_sort(left_array.size(), left_array), merge_sort(right_array.size(), right_array));
